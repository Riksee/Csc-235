BEGIN
INT COMPARE(A, B)
    LENGTH = LEN(A)
    FOR (I = 0; I < LENGTH; I += 1)
        IF (A[I] != B[I])
            RETURN 0
   RETURN 1
END

BEGIN
VOID GLOBAL_TERRORIST()
CREATE INPUT_SUM
OPEN FILE INPUT_FILE from Terrorist.txt
INPUT_SUM = (INT) READLINE 0 from INPUT_FILE
CREATE a MULTIDIMENSIONAL ARRAY of SIZE INPUT_SUM for the dataset called INPUT_FEATURE
CREATE an ARRAY of SIZE INPUT_SUM for the sum of features in the dataset called FEATURE_TOTAL
CREATE an ARRAY of SIZE INPUT_SUM for SIMILARITY_ID
FOR (COUNTER = 1; COUNTER <= INPUT_SUM; COUNTER += 1)
    INPUT_FEATURE[COUNTER - 1] = READLINE COUNTER from INPUT_FILE and SPLIT BY ','
    FEATURE_TOTAL[COUNTER - 1] = SUM OF TERRORIST[COUNTER - 1]
ENDFOR
FOR (I = 0; I < INPUT_SUM; I += 1)
    FOR (J = 0; J < INPUT_SUM; J += 1)
        IF (I == J)
            SKIP
        ENDIF
        IF (FEATURE_TOTAL[I] == FEATURE_TOTAL[J])
            CREATE SIMILAR_FEATURE = COMPARE(INPUT_FEATURE[I], INPUT_FEATURE[J])
            IF (SIMILAR_FEATURE == 1)
                SIMILARITY_ID[I] = J + 1
            ENDIF
        ELSE 
            SIMILARITY_ID[I] = NULL
        ENDIF
    ENDFOR
ENDFOR
CLOSE FILE INPUT_FILE
DISPLAY SIMILARITY_ID
END
